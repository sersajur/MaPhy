\chapter{Практика: Класифікація рівнянь в частинних похідних}
\par\medskip
{\Large\textbf{1. Теоретичні відомості}} \medskip

Маємо рівняння
\begin{equation}
	a_{11}u_{xx} + 2a_{12}u_{xy} + a_{22}u_{yy} + F(x,y,u,u_{x},u_{y}) = 0,
	\label{eq:classpatern}
\end{equation} 
для нього визначаємо дискримінант:
\begin{equation*}
	D \equiv a_{12}^{2} - a_{11}a_{22},
\end{equation*}
при чому дискримінант визначаємо поточково ( $ D = D(x,y) $).
Тоді тип рівняння \eqref{eq:classpatern} визначається в залежності від знаку $ D $:
\begin{itemize}
	\item[] $D > 0$ --- гіперболічного типу в точці $(x,y)$;
	\item[] $D = 0$ --- параболічного типу в точці $(x,y)$;
	\item[] $D < 0$ --- еліптичного типу в точці $(x,y)$.  
\end{itemize}

Характеристичні рівняння для \eqref{eq:classpatern}:
\begin{equation}
	\frac{dy}{dx} = \frac{a_{12} \pm \sqrt{D}}{a_{11}}.
	\label{eq:charact}
\end{equation}

Тоді
\begin{enumerate}
	\item[a)] $D > 0$, 
		$\phi(x,y)=C$ та $\psi(x,y)=C$ --- загальні інтеграли характеристичних рівнянь \eqref{eq:charact}.
		Якщо виконати заміну змінних
		\begin{eqnarray*}
			\xi &=& \phi(x,y),\\
			\eta &=& \psi(x,y), 
		\end{eqnarray*}
		то зведемо рівняння \eqref{eq:classpatern} до першої канонічної гіперболічної форми:
		\begin{equation*}
			u_{\xi\eta} = \Phi(\xi,\eta,u,u_{\xi},u_{\eta}).	 
		\end{equation*}
		Якщо виконати заміну змінних
		\begin{eqnarray*}
			\alpha &=& \frac{\xi + \eta}{2},\\
			\beta &=& \frac{\xi - \eta}{2}, 
		\end{eqnarray*}
		то матимуть місце рівності:
		\begin{eqnarray*}
			u_{\xi} &=& \frac{1}{2}({u_{\alpha} + u_{\beta}}),\\
			u_{\eta} &=& \frac{1}{2}({u_{\alpha} - u_{\beta}}),\\
			u_{\xi\eta} &=& \frac{1}{4}({u_{\alpha\alpha} - u_{\beta\beta}}),
		\end{eqnarray*} 
		і тоді зведемо рівняння \eqref{eq:classpatern} до другої канонічної гіперболічної форми:
		\begin{equation*}
			u_{\alpha\alpha} - u_{\beta\beta} = \Phi_{1}(\alpha,\beta,u,u_{\alpha},u_{\beta}).	 
		\end{equation*}
	\item[б)] $D = 0$,
		$\phi(x,y)=C$ --- єдиний загальний інтеграл характеристичного рівняння \eqref{eq:charact}.
		Виконуючи заміну змінних
		\begin{eqnarray*}
			\xi &=& \phi(x,y),\\
			\eta &=& \psi(x,y), 
		\end{eqnarray*}
		де $ \psi $ --- довільна лінійно незалежна від $ \phi $ функція, зведемо рівняння \eqref{eq:classpatern} до канонічної параболічної форми:
		\begin{eqnarray*}
			u_{\eta\eta} &=& \Phi(\xi,\eta,u,u_{\xi},u_{\eta})\\
			\text{чи}&&\\
			u_{\xi\xi} &=& \Phi(\xi,\eta,u,u_{\xi},u_{\eta}) \text{(якщо взяти $\xi$ і $\eta$ навпаки)}. 
		\end{eqnarray*}
	\item[в)] $D < 0$, 
		$\phi(x,y)=C$ та $\phi^{*}(x,y)=C$ --- комплексно спряжені загальні інтеграли характеристичних рівнянь \eqref{eq:charact}.
		Виконавши заміну змінних
		\begin{eqnarray*}
			\xi &=& \Re\phi,\\
			\eta &=& \Im\phi, 
		\end{eqnarray*}
		зведемо рівняння \eqref{eq:classpatern} до канонічної еліптичної форми:
		\begin{equation*}
			u_{\xi\xi} + u_{\eta\eta} = \Phi(\xi,\eta,u,u_{\xi},u_{\eta}).	 
		\end{equation*}
\end{enumerate}

\par\medskip
{\Large\textbf{2. Приклади}} \medskip

%#1
{\large\textbf{Приклад 1}}

\begin{equation*}
	x^{2}u_{xx} - y^{2}u_{yy} = 0.
\end{equation*}

Для нього:
\begin{eqnarray*}
	a_{11} = x^{2}, a_{12} = 0, a_{22} = -y^{2},& D = a_{12}^{2} - a_{11}a_{22} = x^{2}y^{2},& D > 0, \text{якщо $x\neq0, y\neq0$}.
\end{eqnarray*}

Рівняння характеристик:
\begin{equation*}
	\frac{dy}{dx} = \frac{a_{12} \pm \sqrt{D}}{a_{11}} = \pm \frac{\sqrt{x^{2}y^{2}}}{x^{2}} = \pm \frac{\left|x\right|\cdot\left|y\right|}{x^{2}} = \pm \frac{\left|y\right|}{\left|x\right|}.
\end{equation*}

Область гіперболічності --- це об'єднання чотирьох квадрантів (без характеристичних осей), тому в кожному квадранті можна (і навіть слід) приводити рівняння окремо, розкриваючи відповідно модулі у рівнянні характеристик. Але там всюди стоїть знак "`$\pm$"', тому з точністю до перестановки змінних нема різниці, якщо ці модулі взагалі зняти.
Тому можемо виконати перетворення:
\begin{equation*}
	\frac{dy}{dx} = \pm \frac{y}{x} \Leftrightarrow \frac{dy}{y} = \pm \frac{dx}{x} \Leftrightarrow \ln{\left|y\right|} = \pm \ln{\left|x\right|} + C,\quad C \in \mathbb{R}.
\end{equation*}

Тоді
\begin{equation*}
	\text{або} \quad \ln{\left|xy\right|} = C, \quad \text{або} \quad \ln{\left|\frac{y}{x}\right|} = C,
\end{equation*}
звідки маємо загальні інтеграли
\begin{equation*}
	xy=C \quad \text{та} \quad \frac{y}{x}=C.
\end{equation*}

Поклавши $\xi=xy,\; \eta = \frac{y}{x}$, маємо:
\begin{align*}
	&u_{x} =  u_{\xi}\xi_{x}+u_{\eta}\eta_{x} = u_{\xi}y + u_{\eta}(-\frac{y}{x^{2}}),\\ 
	u_{xx}=&(u_{\xi})_{x}y + u_{\xi}y_{x} + (u_{\eta})_{x}(-\frac{y}{x^{2}}) + u_{\eta}\frac{2y}{x^{3}} =  y(u_{\xi\xi}\xi_{x} + u_{\xi\eta}\eta_{x}) + 0 - \frac{y}{x^{2}}(u_{\eta\xi}\xi_{x} + u_{\eta\eta}\eta_{x}) + u_{\eta}\frac{2y}{x^{3}} = \\
	= &y(u_{\xi\xi}y-u_{\xi\eta}\frac{y}{x^{2}}) - \frac{y}{x^2}(u_{\eta\xi}y - u_{\eta\eta}\frac{y}{x^{2}}) + u_{\eta}\frac{2y}{x^{3}}=u_{\xi\xi}y^{2} + \frac{y^{2}}{x^{4}}u_{\eta\eta}-\frac{2y^{2}}{x^{2}}u_{\xi\eta} + u_{\eta}\frac{2y}{x^{3}}, \\
		u_{y}= &u_{\xi}\xi_{y} + u_{\eta}\eta_{y} = u_{\xi}x + u_{\eta}\frac{1}{x},\\
		u_{yy}= &x(u_{\xi\xi}\xi_{y}+u_{\xi\eta}\eta_{y}) + \frac{1}{x}(u_{\eta\xi}\xi_{y} + u_{\eta\eta}\eta_{y}) = x(u_{\xi\xi}x + u_{\xi\eta}\frac{1}{x}) + \frac{1}{x}(u_{\eta\xi}x + u_{\eta\eta}\frac{1}{x}) =\\
		= &x^{2}u_{\xi\xi}+2u_{\xi\eta} + \frac{1}{x^{2}}u_{\eta\eta}.
\end{align*} 

Тоді, виконавши підстановку у початкове рівняння, приводимо до канонічної форми:
\begin{align*}
	&x^{2}u_{xx}-y^{2}u_{yy} = 0 \; \Leftrightarrow \; x^{2}y^{2}u_{\xi\xi} + \frac{y^{2}}{x^{2}}u_{\eta\eta}-2y^{2}u_{\xi\eta}+\frac{2y}{x}u_{\eta}-x^{2}y^{2}u_{\xi\xi}-2y^{2}u_{\xi\eta}-\frac{y^{2}}{x^{2}}u_{\eta\eta}=0 \; \Leftrightarrow \\
	\Leftrightarrow  \;
	&-4y^{2}u_{\xi\eta}+\frac{2y}{x}u_{\eta}=0 \; \Leftrightarrow \; u_{\xi\eta} - \frac{1}{2xy}u_{\eta} = 0 \; \Leftrightarrow \; u_{\xi\eta} - \frac{1}{2\xi}u_{\eta}=0.
\end{align*}

Можна і іншим способом. Наприклад, поклавши $\alpha = \frac{1}{2}(\xi + \eta) = \frac{1}{2}(xy+\frac{y}{x}),\; \beta = \frac{1}{2}(\xi - \eta) = \frac{1}{2}(xy+\frac{y}{x})$. Але можна загальні інтеграли брати і в їх початковому вигляді $ \ln{\left|y\right|} \pm \ln{\left|x\right|}  = C$. Тоді, поклавши $\xi = \ln{\left|x\right|}, \; \eta = \ln{\left|y\right|}$, маємо:
\begin{align*}
&u_{x} = u_{\xi}\xi_{x} + u_{\eta}\eta_{x} = u_{\xi}\frac{1}{x} + 0, \\
&u_{xx} = -\frac{1}{x^{2}}u_{\xi}+\frac{1}{x}(u_{\xi\xi}\xi_{x}+u_{\xi\eta}\eta_{x}) = - \frac{1}{x^{2}}u_{\xi} + \frac{1}{x^{2}}u_{\xi\xi},\\
&u_{y} = u_{\eta}\frac{1}{y},\\
&u_{yy} = -\frac{1}{y^{2}}u_{\eta} + \frac{1}{y}(u_{\eta\xi}\xi_{y} + u_{\eta\eta}\eta_{y}) = -\frac{1}{y^{2}}u_{\eta} + \frac{1}{y^{2}}u_{\eta\eta}.
\end{align*}

Тоді, виконавши підстановку у початкове рівняння, приводимо до канонічної форми:
\begin{equation*}
	x^{2}u_{xx} - y^{2}u_{yy} = 0 \; \Leftrightarrow \; -u_{\xi} + u_{\xi\xi} + u_{\eta} - u_{\eta\eta} = 0 \; \Leftrightarrow \; u_{\xi\xi} - u_{\eta\eta} + u_{\eta} - u_{\xi} = 0. 
\end{equation*} 

\begin{tabular}{l l l}
\textit{Відповідь:} & гіперболічне при $x^{2}y^{2} > 0$: 	&$u_{\xi\eta}-\frac{1}{2\xi}u_{\eta}=0$ 	(при $\xi = xy, \eta = \frac{y}{x}$) або \\
										& 												&$u_{\xi\xi} - u_{\eta\eta} + u_{\eta} -u_{\xi} = 0$ 		(при $\xi = \ln{\left|x\right|}, \eta = \ln{\left|y\right|}$); \\
										&	параболічне при $x=0$ або $y=0$: 		&$u_{yy}=0$ або $u_{xx}=0$ відповідно;\\
										&при $x=y=0$: &$0=0$ --- нічого класифікувати.\\
\end{tabular}

{\large\textbf{Приклад 2}}

\begin{equation*}
	(1+x^{2})u_{xx} + (1+y^{2})u_{yy} + yu_{y} = 0.
\end{equation*}

Для нього:
\begin{align*}
	&a_{11} = 1 + x^{2}, \quad a_{12} = 0, \quad a_{22} = 1 + y^{2}, \\
	&D = a_{12}^{2} - a_{11}a_{22} = -(1+x^{2})(1+y^{2}) < 0, \quad \text{при $(x;y) \in \mathbb{R}$}.
\end{align*}
Рівняння всюди еліптичне. Рівняння характеристик:
\begin{align*}
	 &\frac{dy}{dx} = \frac{a_{12} \pm \sqrt{D}}{a_{11}} = \frac{\pm i\sqrt{(1+x^{2})(1+y^{2})}}{1+x^{2}} = \pm i\frac{\sqrt{1+y^{2}}}{\sqrt{1+x^{2}}} \Leftrightarrow \\
	\Leftrightarrow \;& \frac{dy}{\sqrt{1+y^{2}}} = \pm i\frac{dx}{\sqrt{1+x^{2}}} \Rightarrow \\
	\Rightarrow \;&\ln{\left|y+\sqrt{1+y^{2}}\right|} = \pm i\ln{\left|x+\sqrt{1+x^{2}}\right|} + C. \\
\end{align*}
 
Беремо дійсну й уявну частини загальних інтегралів:
\begin{align*}
	\left\{ 
	\begin{aligned}
		&\xi=\ln{\left|x+\sqrt{1+x^{2}}\right|} \\
		&\eta=\ln{\left|y+\sqrt{1+y^{2}}\right|}
	\end{aligned},
	\right.
\end{align*}
тоді маємо:
\begin{align*}
	&u_{x} = u_{\xi}\xi{x} + u_{\eta}\eta_{x} = u_{\xi}\frac{1}{\sqrt{1+x^{2}}}, \\
	&u_{xx} = -x(1+x^{2})^{-\frac{3}{2}}u_{\xi} + \frac{1}{\sqrt{1+x^{2}}}(u_{\xi\xi}\xi_{x} + u_{\xi\eta}\eta_{x}) = -x(1+x^{2})^{-\frac{3}{2}}u_{\xi} + \frac{u_{\xi\xi}}{1+x^{2}},\\
	&\text{цілком аналогічно:} \\
	&u_{y} = u_{\eta}\frac{1}{\sqrt{1+y^{2}}}, \\
	&u_{yy} = -y(1+y^{2})^{-\frac{3}{2}}u_{\eta} + \frac{u_{\eta\eta}}{1+y^{2}}.
\end{align*}
Тоді, підставляючи у початкове рівняння, маємо:
\begin{align*}
	&(1+x^{2})u_{xx} + (1+y^{2})u_{yy} + yu_{y} = 0 \Leftrightarrow \\
	\Leftrightarrow \;&-x(1+x^{2})^{-\frac{1}{2}}u_{\xi} + u_{\xi\xi} + u_{\eta\eta} - y(1+y^{2})^{-\frac{1}{2}}u_{\eta}+y(1+y^{2})^{-\frac{1}{2}}u_{\eta} = 0 \Leftrightarrow \\
	\Leftrightarrow \;& u_{\xi\xi} + u_{\eta\eta} - x(1+x^{2})^{-\frac{1}{2}}u_{\xi} = 0,
\end{align*}
оскільки $\xi = \ln{\left|x+\sqrt{1+x^{2}}\right|}= \texttt{arsh}\,{x}$, то $x = \sh{\xi}$, тоді:
\begin{equation*}
	\frac{x}{\sqrt{1+x^{2}}} = \frac{\sh{\xi}}{\sqrt{1+{\sh^{2}{\xi}}}} = \frac{\sh{\xi}}{\ch{\xi}} = \tanh{\xi},
\end{equation*}
а тоді приводимо рівняння до виду:
\begin{equation*}
	u_{\xi\xi} + u_{\eta\eta} - (\tanh{\xi})u_{\xi} = 0.
\end{equation*}

\textit{Відповідь:} еліптичне, $u_{\xi\xi} + u_{\eta\eta} - (\tanh{\xi})u_{\xi} = 0$, (при $\xi = \ln{\left|x+\sqrt{1+x^{2}}\right|}, \; \eta = \ln{\left|y+\sqrt{1+y^{2}}\right|} $).

\medskip
{\large\textbf{Приклад 3}}

\begin{equation*}
	y^{2}u_{xx} + 2yu_{xy} + u_{yy} = 0.
\end{equation*}

Для нього:
\begin{align*}
	&a_{11} = y^{2}, \quad a_{12} = y, \quad a_{22} = 1, \\
	&D = a_{12}^{2} - a_{11}a_{22} = y^{2} - y^{2}\cdot 1 = 0, \quad \text{при $(x;y) \in \mathbb{R}$}.
\end{align*}
Рівняння всюди параболічне. Його рівняння характеристики:
\begin{equation*}
	\frac{dy}{dx} = \frac{a_{12} \pm \sqrt{D}}{a_{11}} = \frac{y}{y^{2}} = \frac{1}{y} \quad \Leftrightarrow\ \quad ydy = dx,
\end{equation*} 
а загальний інтеграл рівняння характеристики:
\begin{equation*}
	y^{2} - 2x = C.
\end{equation*}
Взявши 
\begin{align*}
\left\{
\begin{aligned}
	&\xi = y^{2}-2x \\
	&\eta = y
\end{aligned},
\right.
\end{align*}
маємо:
\begin{align*}
	&u_{x} = u_{\xi}\xi_{x} + u_{\eta}\eta_{x} = -2u_{\xi},\\
	&u_{xx} = -2(u_{\xi\xi}\xi_{x}+u_{\xi\eta}\eta_{x}) = -2(-2u_{\xi\xi}) = 4u_{\xi\xi},\\
	&u_{xy} = -2(u_{\xi\xi}\xi_{y} + u_{\xi\eta}\eta_{y}) = -2(u_{\xi\xi}\cdot 2y + u_{\xi\eta}),\\
	&u_{y} = u_{\xi}\xi_{y} + u_{\eta}\eta_{y} = u_{\xi}\cdot 2y + u_{\eta},\\
	&u_{yy} = 2u_{\xi} + 2y(u_{\xi\xi}\xi_{y}+u_{\xi\eta}\eta_{y}) + u_{\eta\xi}\xi_{y} + u_{\eta\eta}\eta_{y}.
\end{align*}
Підставляючи у початкове рівняння, зводимо його до канонічного виду:
\begin{align*}
	&y^{2}u_{xx} + 2yu_{xy} + u_{yy} = 0 \Leftrightarrow \\
	\Leftrightarrow\ &4y^{2}u_{\xi\xi} - 4y(u_{\xi\xi}2y + u_{\xi\eta}) + 2u_{\xi} + 2y(u_{\xi\xi}2y + u_{\xi\eta}\cdot 1) + u_{\eta\xi}\cdot 2y + u_{\eta\eta}\cdot 1 = 0 \Leftrightarrow\ \\
	\Leftrightarrow\ &4y^{2}u_{\xi\xi} - 8y^{2}u_{\xi\xi}-4yu_{\xi\eta} + 2u_{\xi} + 4y^{2}u_{\xi\xi} + 2yu_{\xi\eta} + 2yu_{\eta\xi} + u_{\eta\eta} = 0 \Leftrightarrow\ u_{\eta\eta} + 2u_{\xi} = 0.
\end{align*}

\textit{Відповідь:} параболічне всюди, $u_{\eta\eta} + 2u_{\xi} = 0$, (при $\xi = y^{2}-2x, \; \eta = y$).

	\item[2)] $u_{xx} + 2u_{xy} - 3u_{yy} + 2u_{x} + 6u_{y} = 0$;
	\item[3)] $yu_{xx} - xu_{yy} + u_{x} + yu_{y} = 0$;
	\item[4)] $u_{xx} + xyu_{yy} = 0$;
	\item[5)] $u_{xx} - 2cos{x}\cdot u_{xy} - (3 + sin^{2}{x})u_{yy} - yu_{y} = 0$.
\end{itemize}

\chapter{Практика: Метод Фур'є розділення змінних}
\par\medskip
{\Large\textbf{1. Приклади}} \medskip

%#1
{\large\textbf{Приклад 1}}

Дано тонкий однорідний стержень $0 < x < l$, бічна поверхня якого ізольована (теплоізоляція). Знайти закон розподілу температури в стержні, якщо кінці стержня підтримуються при нульовій температурі, а початкова температура $\left.u\right|_{t=0} = u_0(x)=Ax(l-x)$, де $A = const$.

Маємо задачу:
\begin{align*}
\left\{
\begin{aligned}
	&u_t = a^2u_{xx}, \quad 0<x<l, \\
	&u(0,t) = u(l,t) = 0, \\
	&u(x,0) = Ax(l-x).
\end{aligned}	
\right.
\end{align*}

Методом Фур'є: $u(x,t) = X(x)T(t)$, тоді
\begin{equation*}
	XT' = a^2TX'' \Leftrightarrow \frac{T'}{a^2T} = \frac{X''}{X} = -\lambda,
\end{equation*}
звідки
\begin{align*}
\left\{
\begin{aligned}
	&X'' + \lambda X = 0, \\
	&T' +\lambda a^2T = 0, 
\end{aligned}	
\right.
\end{align*}
Тоді маємо задачу Штурма-Ліувілля:
\begin{align*}
\left\{
\begin{aligned}
	&X'' + \lambda X = 0, \\
	&X(0) = X(l) = 0, 
\end{aligned}	
\right.
\end{align*}
звідки
\begin{equation*}
	\lambda_n=\left(\frac{\pi n}{l}\right)^2, \quad X_n(x) = \sin{\frac{\pi nx}{l}}.
\end{equation*}
Тоді маємо 
\begin{equation*}
	T_n' + \lambda_n a^2T_n = 0 \quad \Leftrightarrow \quad T_n' + \left(\frac{\pi na}{l}\right)^2T_n = 0 \quad \Rightarrow \quad T_n = C_ne^{-\left(\frac{\pi na}{l}\right)^2t}.
\end{equation*}
Таким чином, маємо:
\begin{equation*}
	u(x,t) = \sum_{n=1}^{\infty}C_ne^{-\left(\frac{\pi na}{l}\right)^2t}\sin{\frac{\pi nx}{l}}.
\end{equation*}
Початкова умова:
\begin{equation*}
	u(x,0) = \sum_{n=1}^{\infty}C_n\sin{\frac{\pi nx}{l}} = Ax(l - x).
\end{equation*}
Розкладаємо $Ax(l-x)$ в ряд Фур'є по системі $\left\{\sin{\frac{\pi nx}{l}}\right\}_{n=1}^{\infty}$. Множимо це на $\sin{\frac{\pi mx}{l}}, \; m \in \mathbb{N}$ і інтегруємо.
\begin{align*}
 &\int\limits_{0}^{l}{\sin{\frac{\pi nx}{l}}\cdot\sin{\frac{\pi mx}{l}}dx} = 
\left\{
\begin{aligned}
	&0, \quad m \neq n, \\
	&\frac{l}{2},\quad m = n, 
\end{aligned} \qquad \text{тоді}
\right.\\
	\frac{l}{2}C_n &= \int\limits_{0}^{l}{Ax(l-x)\sin{\frac{\pi nx}{l}}dx} = -\frac{l}{\pi n}\int\limits_{0}^{l}{Ax(l-x)d\left(\cos{\frac{\pi nx}{l}}\right)} = \\
	&=\frac{l}{\pi n}\left({\left.-Ax(l-x)\cos{\frac{\pi nx}{l}}\right|_{0}^{l} + \int\limits_{0}^{l}{\cos{\frac{\pi nx}{l}}d{\left(Ax(l-x)\right)}}}\right) = \frac{lA}{\pi n}\int\limits_{0}^{l}{(l-2x)\cos{\frac{\pi nx}{l}}dx} = \\
	&=\frac{lA}{\pi n}\cdot\frac{l}{\pi n}\int\limits_{0}^{l}{(l-2x)d\left(\sin{\frac{\pi nx}{l}}\right)} = \frac{Al^2}{\pi^2n^2}\left(\left.\left(l-2x\right)\sin{\frac{\pi nx}{l}}\right|_{0}^{l} + 2\int\limits_{0}^{l}{\sin{\frac{\pi nx}{l}}dx}\right) = \\
	&=\frac{2Al^3}{\pi^3 n^3}\left(\left.-\cos{\frac{\pi nx}{l}}\right|_{0}^{l}\right) = \frac{2Al^3}{\pi^3n^3}\left(1 - (-1)^n\right)=
	\left\{
	\begin{aligned}
		&0, \quad n = 2k, \\
		&\frac{4Al^3}{\pi^3(2k+1)^3}, \quad n = 2k+1.
	\end{aligned}
	\right.
\end{align*}

Значить,
\begin{equation*}
	C_n = 
	\left\{
	\begin{aligned}
		&0, \quad n=2k, \\
		&\frac{8Al^2}{\pi^3(2k+1)^3}, \quad n = 2k+1.
	\end{aligned}
	\right.
\end{equation*}

Тоді маємо: 

\begin{equation*}
	u(x,t) = \frac{8Al^2}{\pi^3}\sum_{k=0}^{\infty}{\frac{1}{(2k+1)^3}e^{-\left(\frac{\pi(2k+1)a}{l}\right)^2t}\cdot\sin{\frac{\pi(2k+1)x}{l}}}.
\end{equation*}

\textit{Відповідь:} $u(x,t) = \frac{8Al^2}{\pi^3}\sum_{k=0}^{\infty}{\frac{1}{(2k+1)^3}e^{-\left(\frac{\pi(2k+1)a}{l}\right)^2t}\sin{\frac{\pi(2k+1)x}{l}}}$.

\textbf{Зауваження:} якщо, наприклад, на кінець $x = l$ подається зовні тепловий потік, то умова пишеться так:
\begin{equation*}
	u_x(l,t) \equiv \left.u_x\right|_{x=l} = q(t);
\end{equation*}
якщо ж $q(t) \equiv 0$, то кінець $x = l$ теплоізольований.
Якщо ж сказано, що на кінці $x = l$ йде теплообмін з оточуючим середовищем, то умова пишеться приблизно так:
\begin{equation*}
	\left.\left(u_x + hu\right)\right|_{x=l}=hu_0,
\end{equation*}
де 	$h$ --- коефіцієнт теплообміну,
		$u_0$ --- температура оточуючого середовища;
якщо сказано, що є вільний теплообмін $(u_0 = 0)$, тоді просто маємо $\left.\left(u_x + hu\right)\right|_{x=l} = 0$.

{\large\textbf{Приклад 2}}

\begin{align*}
\left\{
\begin{aligned}
	&u_t = u_{xx} - 1, \quad 0<x<1, \\
	&u(0,t) = u(1,t) = 0, \\
	&u(x,0) = 2x^2 - 2x.
\end{aligned}	
\right.
\end{align*}

Спочатку розв'язуємо (принаймні, починаємо розв'язувати) задачу з однорідною правою частиною:
\begin{align*}
\left\{
\begin{aligned}
	&u_t = u_{xx}, \quad 0<x<1, \\
	&u(0,t) = u(1,t) = 0. 
\end{aligned}	
\right.
\end{align*}
Розділяючи змінні в однорідній задачі
\begin{equation*}
	u(x,t) = X(x)T(t),
\end{equation*} 
приходимо до задачі Штурма-Ліувілля:
\begin{align*}
\left\{
\begin{aligned}
	&X'' + \lambda X = 0, \\
	&X(0) = X(1) = 0, 
\end{aligned}	
\right.
\end{align*}
звідки
\begin{equation*}
	\lambda_n=\left(\pi n\right)^2, \quad X_n(x) = \sin{(\pi nx)}, \; n \in \mathbb{N}.
\end{equation*}
Тоді розв'язок вихідної неоднорідної задачі шукаємо у вигляді ряда 
\begin{equation*}
	u(x,t) = \sum_{n=1}^{\infty}{T_n(t)X_n(x)}
\end{equation*}
по системі власних функцій $\left\{X_n(x)=\sin(\pi nx)\right\}$ задачі Штурма-Ліувілля для відповідної однорідної задачі. 
Розкладемо праву частину за цією системою функцій: 
\begin{align*}
	-1 = \sum_{n=1}^{\infty}{f_nX_n(x)}, \quad \text{маємо} \quad \int\limits_{0}^{1}{(-1)\sin{(\pi nx)}dx} = f_n\cdot\frac{1}{2}, \quad \text{звідки} \\
	\frac{(-1)^n-1}{\pi n} = f_n\cdot\frac{1}{2} \quad \Leftrightarrow \quad f_n = \frac{2\cdot ((-1)^n-1)}{\pi n}. 
\end{align*}
Тоді, підставивши заготовку в вихідне рівняння, матимемо:
\begin{align*}
	&\sum_{n=1}^{\infty}{T'_n(t)X_n(x) = \sum_{n=1}^{\infty}{T_n(t)X''_n(x)} + \sum_{n=1}^{\infty}{f_nX_n(x)}} \quad \Leftrightarrow\ \\
	\Leftrightarrow \quad &\sum_{n=1}^{\infty}{\left(T'_n(t) + (\pi n)^2T_n(t) - f_n\right)\sin{(\pi nx)}} = 0.
\end{align*}
З початкової умови маємо:
\begin{align*}
	&u(x,0) = \sum_{n=1}^{\infty}{T_n(0)X_n(x)} = 2x^2 - 2x \quad \Leftrightarrow \\
	\Leftrightarrow \quad &T_n(0)\cdot\frac{1}{2} = \int\limits_{0}^{1}{\left(2x^2 - 2x\right)\sin(\pi nx)dx} = 4\frac{(-1)^n - 1}{\pi^3 n^3} \quad \Leftrightarrow\ \\
	&T_n(0) = \frac{8\cdot((-1)^n - 1)}{\pi^3 n^3}.
\end{align*}
Тоді маємо $\forall  n \in \mathbb{N}$ рівняння:
\begin{align*}
\left\{
\begin{aligned}
	&T'_n(t) + (\pi n)^2T_n(t) - f_n = 0, \\
	&T_n(0) = \frac{8\cdot((-1)^n - 1)}{\pi^3 n^3}.
\end{aligned}	
\right.
\end{align*}

Рівняння розв'язуємо методом варіації постійної.
\begin{equation*}
	T'_n(t) = -(\pi n)^2T_n(t) \quad \Leftrightarrow \quad T_n(t) = C_ne^{-(\pi n)^2t}.
\end{equation*}
Припускаємо, що $C_n = C_n(t)$, тоді $T_n(t) = C_n(t)e^{-(\pi n)^2t}$, звідки
\begin{align*}
	C'_n(t)e^{-(\pi n)^2t} - C_n(t)(\pi n)^2e^{-(\pi n)^2t} + (\pi n)^2C_n(t)e^{-(\pi n)^2t} - f_n = 0 \quad \Leftrightarrow\ \quad C'_n(t)e^{-(\pi n)^2t} = f_n, \text{тоді} \\
	C'_n(t) = f_n\cdot e^{(\pi n)^2t} \quad \Leftrightarrow \quad C_n(t) = \frac{f_n}{(\pi n)^2}e^{(\pi n)^2t} + C.
\end{align*}
Значить, загальний розв'язок неоднорідного звичайного ДР:
\begin{equation*}
	T_n(t) = C_n(t)e^{-(\pi n)^2t} = \frac{f_n}{(\pi n)^2} + Ce^{-(\pi n)^2t}.
\end{equation*}
З початкової умови:
\begin{equation*}
	\frac{8\cdot ((-1)^n - 1)}{(\pi n)^3} = \frac{2\cdot((-1)^n - 1)}{(\pi n)^3} + C\cdot 1 \quad \Leftrightarrow \quad C = \frac{6\cdot((-1)^n-1)}{(\pi n)^3}.
\end{equation*}
Значить, 
\begin{equation*}
	T_n(t) = \frac{2\cdot((-1)^n-1)}{(\pi n)^3} + \frac{6\cdot((-1)^n-1)}{(\pi n)^3}e^{-(\pi n)^2t}.
\end{equation*}
При $n = 2k$ буде $T_n(t) = 0$, тому розглядаємо при $n = 2k+1$.
Тоді остаточно:


\begin{equation*}
			u(x,t) = \sum_{k=0}^{\infty}{\left(\frac{-4}{(\pi(2k+1))^3} - \frac{12}{(\pi(2k+1))^3}e^{-(\pi(2k+1))^2t}\right)\sin{(\pi(2k+1)x)}} \quad \text{або ж}
\end{equation*}
\begin{equation*}
	u(x,t) = \frac{-4}{\pi^3} \sum_{k=0}^{\infty}{\left(1 + 3e^{-(\pi(2k+1))^2t}\right)\frac{\sin{(\pi(2k+1)x)}}{(2k + 1)^3}}.\\
\end{equation*}

\textit{Відповідь:} $u(x,t) = \frac{-4}{\pi^3}\sum_{k=0}^{\infty}{\left(1 + 3e^{-(\pi(2k+1))^2t}\right)\frac{\sin{(\pi(2k+1)x)}}{(2k + 1)^3}}$.

{\large\textbf{Приклад 3}}

\begin{align*}
\left\{
\begin{aligned}
	&u_t = u_{xx} + x, \quad 0<x<1, \; t > 0,\\
	&u(0,t) = t, \; u(1,t) = 2t + 1, \\
	&u(x,0) = 2x^2 - x.
\end{aligned}	
\right.
\end{align*}

Шукаємо розв'язок у вигляді 
\begin{equation*}
	u(x,t) = v(x,t) + U(x,t),
\end{equation*}
де $v(x,t)$ --- нова невідома функція,
a $U(x,t)$ --- підібрана так, щоб задовольняти неоднорідні граничні умови, тобто $U(0,t) = t, \; U(1,t) = 2t + 1$. 
Якщо треба таку $U(x,t)$, щоб $U(0,t) = \mu_1(t), \; U(l,t) = \mu_2(t)$, то можна взяти 
\begin{equation*}
	U(x,t) = \mu_1(t) + (\mu_2(t) - \mu_1(t))\frac{x}{l}.
\end{equation*}
Тоді, в  нашому випадку отримуємо: 
\begin{equation*}
	U(x,t) = t + (2t+1-t)\frac{x}{1} = t + (t+1)x.
\end{equation*}
Значить, маємо представлення
\begin{equation*}
	u(x,t) = v(x,t) + t + (t+1)x.
\end{equation*}
Підставляємо в задачу:
\begin{equation*}
	u_t = u_{xx} + x \quad \Leftrightarrow\ \quad v_t + 1 + x = v_{xx} + x \quad \Leftrightarrow \quad v_t = v_{xx} -1.
\end{equation*}
Граничні умови:
\begin{align*}
	&u(0,t) = t \quad \Leftrightarrow \quad v(0,t) + t + 0 = t \quad \Leftrightarrow \quad v(0,t) = 0; \\
	&u(1,t) = 2t + 1 \quad \Leftrightarrow \quad v(1,t) + t + t + 1 = 2t + 1 \quad \Leftrightarrow \quad v(1,t) = 0.
\end{align*}
Початкова умова:
\begin{equation*}
	u(x,0) = 2x^2 - x \quad \Leftrightarrow\ \quad v(x,0) + x = 2x^2 - x \quad \Leftrightarrow \quad v(x,0) = 2x^2 - 2x.
\end{equation*}
Значить, маємо нову задачу відносно $v(x,t)$, але з однорідними граничними умовами:
\begin{align*}
\left\{
\begin{aligned}
	&v_t = v_{xx} - 1, \quad 0<x<1, \; t > 0,\\
	&v(0,t) = v(1,t) = 0, \\
	&v(x,0) = 2x^2 - 2x.
\end{aligned}	
\right.
\end{align*}
Ця задача була розв'язана в попередньому прикладі, тому можемо відразу виписати розв'язок:
\begin{equation*}
	u(x,t) = t + (t+1)x - \frac{4}{\pi^3} \sum_{k=0}^{\infty}{\left(1 + 3e^{-(\pi(2k+1))^2t}\right)\frac{\sin{(\pi(2k+1)x)}}{(2k+1)^3}}.
\end{equation*}

\textit{Відповідь:} $u(x,t) = t + (t+1)x - \frac{4}{\pi^3} \sum_{k=0}^{\infty}{\left(1 + 3e^{-(\pi(2k+1))^2t}\right)\frac{\sin{(\pi(2k+1)x)}}{(2k+1)^3}}$.

\textbf{Зауваження:} при інших неоднорідних граничних умовах можливі інші функції заміни $U(x,t)$. Наприклад, 
\begin{align*}
	&\text{для} \quad u_x(0,t)=\alpha(t), \; u_x(l,t) = \beta(t) \quad \text{підійде заміна} \quad U(x,t) = \alpha(t)x + (\beta(t) - \alpha(t))\frac{x^2}{2l};\\
	&\text{при} \quad u(0,t) = \alpha(t), \; u_x(l,t)=\beta(t) \quad \text{можна брати} \quad U(x,t) = \alpha(t) + \beta(t)\frac{x^2}{2l} \quad \text{і т.д.}
\end{align*}	
У випадку більш загальних неоднорідних умов в якості функції $U(x,t)$, що задовольняє граничні умови, можна спробувати взяти $U(x,t) = a(t) + b(t)x + c(t)x^2$, а потім визначити відповідні коефіцієнти $a(t), b(t), c(t)$. Нам підійде довільна така $U(x,t)$, лише б вона задовольняла граничні умови.

{\large\textbf{Приклад 4}}

Знайти форму $u(x,t)$ закріпленої на кінці $x = 0$ однорідної струни, правий кінець має горизонтальну дотичну, і на струну діє зовнішня сила з щільністю $F(x,t) \neq 0$. В початковий момент $t=0$ форма струни визначається через $\phi(x)$, а швидкість --- через $\psi(x)$. Знайти $u(x,t)$ за умов $\phi(x) = \psi(x) = 0$, а $F(x,t) = a^2$.

Загальна постановка задачі:
\begin{align*}
\left\{
\begin{aligned}
	&\frac{\partial^2{u}}{\partial{t^2}} = a^2\frac{\partial^2{u}}{\partial{x^2}} + F(x,t), \quad 0<x<l, \; t > 0,\\
	&u(0,t) = 0, \; \frac{\partial{u}}{\partial{x}}(l,t) = 0, \\
	&u(x,0) = \phi(x), \; \frac{\partial{u}}{\partial{t}}(x,0) = \psi(x).
\end{aligned}	
\right.
\end{align*}

Знайдемо спочатку власні функції однорідного рівняння 
\begin{align*}
\left\{
\begin{aligned}
	&u_{tt} = a^2u_{xx},\\
	&u(0,t) = u_x(l,t) = 0. 
	\end{aligned}	
\right.
\end{align*}
Покладемо
\begin{equation*}
	u(x,t) = X(x)\cdot T(t),
\end{equation*}
тоді, розділяючи змінні, маємо:
\begin{equation*}
	\frac{T''}{a^2T} = \frac{X''}{X} = -\lambda,
\end{equation*}
звідки 
\begin{align*}
\left\{
\begin{aligned}
	&X'' + \lambda X= 0,\\
	&X(0) = X'(l) = 0. 
	\end{aligned}	
\right.
\end{align*}
Знаходимо власні функції і значення:
\begin{equation*}
	\lambda_k = \left(\frac{\pi (2k-1)}{2l}\right)^2, \quad X_k(x) = \sin{\frac{\pi(2k-1)x}{2l}}, \; k \in \mathbb{N}.
\end{equation*}

Розкладемо $F(x,t)$ по $\left\{X_k(x)\right\}$:
\begin{align*}
	&F(x,t) = \sum_{k=1}^{\infty}{f_k(t)\sin{\frac{\pi (2k - 1)x}{2l}}}, \quad
	\text{тоді} \quad \int\limits_{0}^{l}{F(x,t)\sin{\frac{\pi(2k-1)x}{2l}}dx} = f_k(t)\cdot\frac{l}{2},\\
	\text{бо} \quad &\int\limits_{0}^{l}{\sin^{2}{\frac{\pi(2k-1)x}{2l}}dx} = \frac{l}{2},\quad
	\text{звідки} \quad f_k(t) = \frac{2}{l}\int\limits_{0}^{l}{F(x,t)\sin{\frac{\pi(2k-1)x}{2l}}dx}.
\end{align*}

Шукаємо розв'язок неоднорідного рівняння у вигляді 
\begin{equation*}
	u(x,t) = \sum_{k=1}^{\infty}{u_k(t)X_k(x)} = \sum_{k=1}^{\infty}{u_k(t)\sin{\frac{\pi(2k-1)x}{2l}}}.
\end{equation*}
Підставивши його в рівняння $u_{tt} = a^2u_{xx} + F(x,t)$, маємо: 
\begin{equation*}
	\sum_{k=1}^{\infty}{\left(u''_k(t) + \frac{\pi^2a^2(2k-1)^2}{4l^2}u_k(t) - f_k(t)\right)\sin{\frac{\pi(2k-1)x}{2l}}} = 0,
\end{equation*}
звідки маємо рівняння:
\begin{equation*}
	u''_k(t) + \frac{\pi^2a^2(2k-1)^2}{4l^2}u_k(t) = f_k(t).
\end{equation*}
Початкові  умови для нього будуть такі:
\begin{align*}
	&u(x,0) = \sum_{k=1}^{\infty}{u_k(0)\sin{\frac{\pi(2k-1)x}{2l}}} \quad \Leftrightarrow \quad u_k(0) = \frac{2}{l}\int\limits_{0}^{l}{\phi(x)\sin{\frac{\pi(2k-1)x}{2l}}dx}, \quad \text{аналогічно}\\
	&u_t(x,0) = \sum_{k=1}^{\infty}{u'_k(0)\sin{\frac{\pi(2k - 1)x}{2l}}} = \psi(x) \quad \Leftrightarrow \quad u'_k(0) = \frac{l}{2}\int\limits_{0}^{l}{\psi(x)\sin{\frac{\pi(2k-1)x}{2l}}dx}.
\end{align*}
Після розв'язання ЗДР і знаходження $u_k(t) \; \forall k \in \mathbb{N}$, розв'язок всієї задачі дається через 
\begin{equation*}
	u(x,t) = \sum_{k=1}^{\infty}{u_k(t)X_k(x)} = \sum_{k=1}^{\infty}{u_k(t)\sin{\frac{\pi(2k-1)x}{2l}}}.
\end{equation*}

Знайдемо тепер розв'язок для конкретних $\phi(x) = \psi(x) = 0, \; F(x,t) = a^2$.
Тоді %
\begin{align*}
	f_k(t) 	&= \frac{2}{l}\int\limits_{0}^{l}{F(x,t)\sin{\frac{\pi(2k-1)x}{2l}}dx} = \frac{2}{l}\int\limits_{0}^{l}{a^2\sin{\frac{\pi(2k-1)x}{2l}}dx} = \\
					&=-\frac{2a^2}{l}\cdot\frac{2l}{\pi(2k-1)}\left.\cos{\frac{\pi(2k-1)x}{2l}}\right|_{0}^{l} = \frac{4a^2}{\pi(2k-1)}.
\end{align*}
Значить, треба знайти розв'язки рівнянь:
\begin{equation*}
	u''_k(t) + \frac{\pi^2a^2(2k-1)^2}{4l^2}u_k(t) = \frac{4a^2}{\pi(2k-1)}.
\end{equation*}
Оскільки $\phi(x) = \psi(x) = 0$, то $u_k(0) = u'_k(0) = 0$. Маємо диференціальне рівняння з постійними коефіцієнтами й вільним членом. Його характеристичне рівняння має комплексні корені, можемо пошукати частинний розв'язок неоднорідного рівняння у вигляді константи:
\begin{equation*}
	\tilde{u}_k(t) = \gamma_k \quad \Leftrightarrow \quad \gamma_{k} = \frac{16l^2}{\pi^3(2k-1)^3}.
\end{equation*}
Тому загальний розв'язок неоднорідного запишеться як 
\begin{equation*}
	u_k(t) = \alpha_k\cos{\frac{\pi a(2k-1)t}{2l}} + \beta_k\sin{\frac{\pi a(2k-1)t}{2l}} + \frac{16l^2}{\pi^3(2k-1)^3}, \; k \in \mathbb{N}.
\end{equation*}
З умов $u_k(0) = u'_k(0) = 0$ маємо:
\begin{align*}
	\alpha_k + \frac{16l^2}{\pi^3(2k-1)^3} = 0 \quad \Rightarrow \quad &\alpha_k = -\frac{16l^2}{\pi^3(2k-1)^3}, \\
	\beta_k\frac{\pi a(2k-1)}{2l} = 0 \quad \Rightarrow \quad &\beta_k = 0.
\end{align*}
Таким чином, маємо:
\begin{equation*}
	u_k(t) = \frac{16l^2}{\pi^3(2k-1)^3}\left(1 - \cos{\frac{\pi a(2k-1)t}{2l}}\right) = \frac{32l^2}{\pi^3(2k-1)^3}\sin^2{\frac{\pi a(2k-1)t}{4l}}.
\end{equation*}
Тоді
\begin{equation*}

Видно, що корені цього рівняння строго додатні, їх зліченна кількість і вони зростають. Позначимо їх через $\lambda_n, \; n \in \mathbb{N}$. Відповідно, власні функції $X_n(x) = \sin{\sqrt{\lambda_n}x}$.

\end{equation*}

\begin{equation*}
	u(x,t) = \sum_{n=1}^{\infty}{C_ne^{-a^2\lambda_nt}\sin{\left(x\sqrt{\lambda_n}\right)}}.
\end{equation*}

\begin{align*}

	&\begin{aligned}

	\end{aligned}\\
	\Rightarrow \quad &C_n = \frac{2\sqrt{\lambda_n}}{\sqrt{\lambda_n} - \sin{\sqrt{\lambda_n}}\cos{\sqrt{\lambda_n}}}\int\limits_{0}^{1}{\phi(x)\sin{\left(x\sqrt{\lambda_n}\right)}dx}.
\end{align*} 

\textit{Відповідь:} $u(x,t) = \sum_{n=1}^{\infty}{C_ne^{-a^2\lambda_nt}\sin{\left(x\sqrt{\lambda_n}\right)}}$, де $C_n$ визначено формулою вище, а $\{\lambda_n\}, n \in \mathbb{N}$ --- корені рівняння $\tg{\sqrt{\lambda}} = -\frac{\sqrt{\lambda}}{h}$.

Знайти поперечні коливання прямокутної мембрани $\{0 \leq x \leq p, \; 0 \leq y \leq q\}$ із закріпленими краями, спричинені початковим відхиленням $\phi(x,y)$ та початковою швидкістю $\psi(x,y)$.

\begin{align*}
	\left\{
	\begin{aligned}
		&u_{tt} = a^2(u_{xx} + u_{yy}),\\
		&u(0,y,t) = u(p,y,t) = u(x,0,t) = u(x,q,t) = 0, \\
		&u(x,y,0) = \phi(x,y), \\
		&u_t(x,y,0) = \psi(x,y).
	\end{aligned}
	\right.
\end{align*}		

Розділимо змінні:
\begin{equation*}
	u(x,y,t) = T(t)V(x,y).
\end{equation*}
Тоді
\begin{equation*}
	V(x,y)T''(t) = a^2T(t)\Delta V(x,y) \quad \Leftrightarrow \quad \frac{T''(t)}{a^2T(t)} = \frac{\Delta V(x,y)}{V(x,y)} = -\lambda.
\end{equation*}

Розглянемо 
\begin{equation*}
	\frac{\Delta V}{V} = -\lambda \quad \Rightarrow \quad  V_{xx} + V_{yy} = -\lambda V.
\end{equation*}
Покладемо $V(x,y) = X(x)Y(y)$, тоді
\begin{align*}
	X''Y + XY'' = -\lambda XY \; \Leftrightarrow \; \frac{X''}{X} + \frac{Y''}{Y} = -\lambda \; \Leftrightarrow \; \frac{X''}{X} = -\frac{Y''}{Y}  -\lambda \equiv -\mu \; \Leftrightarrow \; 
	\left\{
	\begin{aligned}
		&X'' + \mu X = 0, \\
		&Y'' + \nu Y = 0,
	\end{aligned}
	\right.
\end{align*}
де $\nu = \lambda - \mu$.

\begin{equation*}
	X(x) = C_1\sin{\sqrt{\mu}x} + C_2\cos{\sqrt{\mu}x},
\end{equation*}
з граничних умов $X(0) = X(p) = 0$ знаходимо:
\begin{equation*}
	\mu_n = \left(\frac{\pi n}{p}\right)^2, \quad X_n(x) = \sin{\frac{\pi nx}{p}}.
\end{equation*}
Аналогічно
\begin{equation*}
Y'' + \nu Y = 0 \quad \Rightarrow \quad \nu_k = \frac{\pi k}{q}, \quad Y_k(y) = \sin{\frac{\pi ky}{q}}.
\end{equation*}
Тоді 
\begin{equation*}
	\lambda_{nk} = \nu_k + \mu_n = \pi^2\left(\frac{n^2}{p^2} + \frac{k^2}{q^2}\right), \quad V_{nk}(x,y) = X_n(x)Y_k(y) = \sin{\frac{\pi nx}{p}}\cdot\sin{\frac{\pi ky}{q}}.
\end{equation*}

Дано, $T''_{nk} + a^2\lambda_{nk}T_{nk} = 0$, звідки маємо:
\begin{equation*}
	T_{nk}(t) = C_{nk}\sin{\left(a\sqrt{\lambda_{nk}}t\right)} + D_{nk}\cos{\left(a\sqrt{\lambda_{nk}}t\right)}.
\end{equation*}

Тоді
\begin{equation*}
	u(x,y,t) = \sum_{n,k=1}^{\infty}{T_{nk}(t)V_{nk}(x,y)} = \sum_{n=1}^{\infty}{\sum_{k=1}^{\infty}{\left(C_{nk}\sin{\left(a\sqrt{\lambda_{nk}}t\right)} + D_{nk}\cos{\left(a\sqrt{\lambda_{nk}}t\right)}\right)\sin{\frac{\pi nx}{p}}\sin{\frac{\pi ky}{q}}}}.
\end{equation*} 
З початкових умов маємо:
\begin{align*}
	u(x,y,0) &= \sum_{n,k=1}^{\infty}{D_{nk}\sin{\frac{\pi nx}{p}}\sin{\frac{\pi ky}{q}}} = \phi(x,y) \quad \Rightarrow \\
	&\Rightarrow \quad	D_{nk} = \frac{4}{pq}\int\limits_{0}^{p}{\int\limits_{0}^{q}{\phi(x,y)\sin{\frac{\pi nx}{p}}\sin{\frac{\pi ky}{q}}dx}dy}, 
\end{align*}
аналогічно
\begin{align*}
	u_t(x,y,0) &= \sum_{n,k = 1}^{\infty}{C_{nk}a\sqrt{\lambda_{nk}}\sin{\frac{\pi nx}{p}}\sin{\frac{\pi ky}{q}}} = \psi(x,y) \quad \Rightarrow \\
	&\Rightarrow \quad C_{nk} = \frac{4}{pqa\sqrt{\lambda_{nk}}}\int\limits_{0}^{p}{\int\limits_{0}^{q}{\psi(x,y)\sin{\frac{\pi nx}{p}}\sin{\frac{\pi ky}{q}}dx}dy}.
\end{align*}
Підставляючи $C_{nk}$ та $D_{nk}$ у $u(x,y,t)$, отримаємо відповідь.

\textit{Відповідь:} $u(x,y,t) = \sum_{n=1}^{\infty}{\sum_{k=1}^{\infty}{\left(C_{nk}\sin{\left(a\sqrt{\lambda_{nk}t}\right)} + D_{nk}\cos{\left(a\sqrt{\lambda_{nk}t}\right)}\right)\sin{\frac{\pi nx}{p}}\sin{\frac{\pi ky}{q}}}}$, 
	де $C_{nk}$ та $D_{nk}$ визначені формулами вище.

\par\medskip
{\Large\textbf{1. Теоретичні відомості}} \medskip

Рівняння 
\begin{equation*}
	x^2y'' + xy' + (x^2-v^2)y = 0
\end{equation*}
називають рівнянням Бесселя, а довільний його розв'язок (не нульовий) --- циліндричною функцією.

\begin{wrapfigure}[7]{l}{150pt}%
	\includegraphics[scale = 0.2]{bessel.png}%
\end{wrapfigure}

Прикладом циліндричної функції може слугувати наступна функція:
\begin{equation*}
	J_v(x) = \sum_{k = 0}^{\infty}{\frac{(-1)^k}{\Gamma(k+v+1)\Gamma(k+1)}\left(\frac{x}{2}\right)^{2k+v}}.
\end{equation*} 
Це так звана функція Бесселя першого роду порядку $v$.

Властивості $J_v(x)$:
\begin{enumerate}
	\item[1)] якщо $\mu_1$ та $\mu_2$ --- дійсні корені рівняння 
		\begin{equation*}
			\alpha J_v(\mu) + \beta\mu J'_v(\mu) = 0, \quad \alpha \geq 0, \; \beta \geq 0, \; \alpha + \beta > 0,
		\end{equation*}
		то при $v > -1$ буде
		\begin{equation*}
			\int\limits_{0}^{1}{xJ_v(\mu_1x)J_v(\mu_2x)dx} = 0,
		\end{equation*}
		якщо $\mu_1^2 \neq \mu_2^2$.
		
		Крім того,
		\begin{equation*}
			\int\limits_{0}^{1}{xJ_v^2(\mu_ix)dx} = \frac{1}{2}\left(J'_v(\mu_i)\right)^2 + \frac{1}{2}\left(1 - \frac{v^2}{\mu_i^2}\right)J_v^2(\mu_i);
		\end{equation*}
	\item[2)] у попередньому співвідношенні $\alpha J_v(\mu) + \beta\mu J'_v(\mu) = 0$ за тих же умов корені всі дійсні й прості (крім, можливо, нуля). Значить, можна перенумерувати додатні корені:
		\begin{equation*}
			\mu_1 < \mu_2 < \ldots
		\end{equation*}
	\item[3)] за тих же умов система $\left\{J_v(\mu_k^{(v)}x)\right\}, \; k \in \mathbb{N}$, утворює повну ортогональну систему в просторі $L_2(0;1)$ з вагою $\rho(x) = x$.
	\item[4)] $J_{v+1}(x) + J_{v-1}(x) = \frac{2v}{x}J_v(x); \quad J'_{v}(x) = J_{v-1}(x) - \frac{v}{x}J_v(x) = \frac{v}{x}J_v(x) - J_{v+1}(x)$;
	\item[5)] якщо $J_v(\mu_i) = 0$ то $\int\limits\limits_{0}^{1}{xJ_v^2(\mu_ix)dx} = \frac{1}{2}J_{v+1}^2(\mu_i)$.
\end{enumerate}

\par\medskip
{\Large\textbf{2. Приклади}} \medskip

%#1
{\large\textbf{Приклад 1}}

Знайти розподіл температури в нескінченному круглому циліндрі $0 \leq r \leq r_0$ при умові, що початкова температура $u(r,0) = u_0(1 - \frac{r^2}{r_0^2})$, а на поверхні циліндра підтримується нульова температура.

Нескінченність циліндра --- не залежить від $z$, однорідність --- не залежить від полярного кута $\phi$. В циліндричних координатах:
\begin{equation*}
	\Delta u = u_{rr} + \frac{1}{r}u_r + \frac{1}{r^2}u_{\phi\phi} + u_{zz}.
\end{equation*}
Тоді
\begin{align*}
	\left\{
	\begin{aligned}
		&u_t = a^2\left(u_{rr} + \frac{1}{r}u_r\right), \; 0 < r < r_0, \\
		&u(r,0) = u_0\left(1 - \frac{r^2}{r_0^2}\right), \; u(r_0, t) = 0.
	\end{aligned}
	\right.
\end{align*}

\begin{align*}
	&u(r,t) = R(r)T(t) \quad \Rightarrow \quad R(r)T'(t) = a^2\left(R''(r)T(t) + \frac{1}{r}R'(r)T(t)\right), \; \text{звідки} \\
	&\frac{T'}{Ta^2} = \frac{R''(r)}{R(r)} + \frac{1}{r}\cdot\frac{R'(r)}{R(r)} = -\lambda \quad \Leftrightarrow \quad 
	\left\{
	\begin{aligned}
		&T' + a^2\lambda T = 0, \\
		&R'' + \frac{1}{r}R' + \lambda R = 0.
	\end{aligned}
	\right.
\end{align*}
З граничних умов маємо $R(r_0) = 0$.

\begin{equation*}
	R'' + \frac{1}{r}R' + \lambda R = 0 \quad \Leftrightarrow \quad r^2R'' + rR' + \lambda r^2R = 0.
\end{equation*}
Зробимо заміну $\tilde{r} = \sqrt{\lambda}r$. Тоді, позначивши $\tilde{R}{(\tilde{r})} \equiv R(r)$, маємо:
\begin{equation*}
	R' = \frac{dR}{dr} = \frac{dR}{d\tilde{r}}\cdot\frac{d\tilde{r}}{dr} = \sqrt{\lambda}\tilde{R}', \quad R'' = \lambda\tilde{R}'',
\end{equation*}
і тоді буде
\begin{equation*}
	\tilde{r}^2\tilde{R}'' + \tilde{r}\tilde{R}' + \tilde{r}^2\tilde{R} = 0,
\end{equation*}
його частинний розв'язок
\begin{equation*}
	\tilde{R}(\tilde{r}) = J_0(\tilde{r}) = J_0\left(\sqrt{\lambda}r\right).
\end{equation*}
Значить, рівнянню $r^2R'' + rR' + \lambda r^2R = 0$ задовольняє функція:
\begin{equation*}
	R(r) = J_0\left(\sqrt{\lambda}r\right).
\end{equation*}
Гранична умова
\begin{equation*}
	R(r_0) = 0 \quad \Leftrightarrow \quad J_0\left(r_0\sqrt{\lambda}\right) = 0 \quad \Leftrightarrow \quad r_0\sqrt{\lambda} = \mu_n, \; n \in \mathbb{N},
\end{equation*}
де $\mu_n$ --- корінь функції Бесселя $J_0(x)$: $J_0(\mu_n) = 0$. 
Значить, 
\begin{equation*}
	\lambda = \lambda_n = \left(\frac{\mu_n}{r_0}\right)^2,
\end{equation*}
і відповідні їм власні функції:
\begin{equation*}
	R_n(r) = J_0\left(\frac{\mu_n}{r_0}r\right).
\end{equation*}
\begin{equation*}
	T'_n + a^2\lambda_nT_n = 0 \quad \Leftrightarrow \quad T_n(t) = C_ne^{-a^2\frac{\mu_n^2}{r_0^2}t}.
\end{equation*}
Значить,
\begin{align*}
	u(r,t) = \sum_{n=1}^{\infty}{C_ne^{-a^2\frac{\mu_n^2}{r_0^2}t}\cdot J_0\left(\frac{r\mu_n}{r_0}\right)}, \\
	u(r,0) = \sum_{n=1}^{\infty}{C_nJ_0\left(\frac{r\mu_n}{r_0}\right)} = u_0\left(1 - \frac{r^2}{r_0^2}\right).
\end{align*}
Тоді
\begin{equation*}
	C_n\int\limits\limits_{0}^{r_0}{r\cdot J_0^2\left(\mu_n\frac{r}{r_0}\right)dr} = \int\limits\limits_{0}^{r0}{u_0\left(1 - \frac{r^2}{r_0^2}\right)r\cdot J_0\left(\mu_n\cdot\frac{r}{r_0}\right)dr}.
\end{equation*}
Тоді
\begin{equation*}
	\int\limits\limits_{0}^{r0}{r\cdot J_0^2\left(\mu_n\frac{r}{r_0}\right)dr} = \left|r=r_0x\right| = r_0^2\int\limits\limits_{0}^{1}{xJ_0^2(\mu_nx)dx} = \frac{r_0^2}{2}J_1^2(\mu_n).
\end{equation*}
Для обчислення другого інтегралу скористаємося табличною формулою:
\begin{equation*}
	\int\limits\limits_{0}^{1}{J_0(\mu_nx)x\left(1-x^2\right)dx} = \frac{2}{\mu_n^2}J_2(\mu_n).
\end{equation*}
Тоді
\begin{align*}
	\int\limits\limits_{0}^{r_0}{u_0\left(1 - \left(\frac{r}{r_0}\right)^2\right)r\cdot J_0\left(\mu_n\cdot\frac{r}{r_0}\right)dr} &= \left|r=r_0x\right| = u_0\cdot r_0^2\int\limits\limits_{0}^{1}{x(1 - x^2)J_0(\mu_nx)dx} = \\
	&= u_0r_0^2\frac{2}{\mu_n^2}J_2(\mu_n) = u_0r_0^2\cdot\frac{2}{\mu_n^2}\cdot\frac{2}{\mu_n}\cdot J_1(\mu_n),
\end{align*}
звідки
\begin{equation*}
	C_n = \frac{8u_0}{\mu_n^3}\cdot\frac{1}{J_1(\mu_n)}.
\end{equation*}

\textit{Відповідь:} $u(r,t) = 8u_0\sum\limits_{n=1}^{\infty}{\frac{1}{\mu_n^3\cdot J_1(\mu_n)}\cdot e^{-\frac{a^2\mu_n^2}{r_0^2}t}\cdot J_0\left(\frac{r}{r_0}\mu_n\right)}$, де $\{\mu_n\}, \; n \in \mathbb{N}$ --- корені рівняння $J_0(\mu)=0$.

\par\medskip
{\Large\textbf{3. Теоретичні відомості (продовження)}} \medskip

Часто використовують \emph{циліндричні функції другого роду (функції Неймана)}
\begin{equation*}
	N_v(x) = \frac{\cos{(\pi v)}\cdot J_v(x) - J_{-v}(x)}{\sin{(\pi v)}}, \; v \notin \mathbb{Z}; \quad N_n(x) \equiv \lim_{v\to n,\;v \notin \mathbb{Z}}{N_v(x)}, \; n \in \mathbb{Z}.
\end{equation*}
Визначник Вронського 
\begin{equation*}
	W\left[J_v(x);N_v(x)\right] = \frac{2}{\pi x} > 0 \quad \forall x > 0,
\end{equation*}
значить $J_v(x)$ та $N_v(x)$ --- лінійно незалежні. Тоді загальний розв'язок рівняння Бесселя можна представити як
\begin{equation*}
	y(x) = C_1J_v(x) + C_2N_v(x),
\end{equation*}
де $C_i = const$. 
$N_0(x) \sim \ln{x}$ та $N_v(x) \sim x^{-v}$ при $x \to 0$, тому в нулі $x = 0$ функція Неймана має особливість.

\emph{Циліндричні функції третього роду (функції Ханкеля)}:
\begin{align*}
	H_v^{(1)}(x) = J_v(x) + iN_v(x), \\
	H_v^{(2)}(x) = J_v(x) - iN_v(x).
\end{align*}
Видно, що $J_v$ та $N_v$ є аналогами $\cos{x}$ та $\sin{x}$, а $H_v^{(1)}$ та $H_v^{(2)}$ --- аналогами $e^{ix}$ та $e^{-ix}$.
